wpa_supplicant.conf,dhcpcd.confの２つでSSIDの名前をSSID-...となっているところのSSIDはつけない

WIFIネットワークがWPA2の場合は、wpa_supplicantの
network={
#
# SSIDは""で囲う
ssid="SSIDを記入"
#
# 平文のパスは""で囲う
psk="パスフレーズ"
#
# WPA-PSK,WPA-EAP,NONE
key_mgmt=WPA-PSK
#
# WPAの場合はWPAに変更する、WPA2の場合はRSN
proto=RSN
#
# ステルスモードの場合は scan_ssid=1 に変更
scan_ssid=0(wifiアクセスポイントが見えるように設定しているとき)

scan_ssid=1(wifiアクセスポイントが見えないに設定しているとき)

・raspberrypiでVPNserverを構築する
raspberrypiでVPNserverを構築する
#SoftEther VPN
auto br0
iface br0 inet static
address 192.168.xxx.xxx # 固定 IP アドレスを設定
netmask 255.255.255.0
network 192.168.xxx.0
broadcast 192.168.xxx.255
gateway 192.168.xxx.xxx　　　　　　　# 使用するルータの IP アドレスを設定
bridge_ports eth0
bridge_maxwait 10
